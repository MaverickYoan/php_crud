CREATE TABLE jeux (
    id SERIAL PRIMARY KEY,
    jeu VARCHAR(255) NOT NULL,
    genre VARCHAR(100) NOT NULL,
    annee INT NOT NULL,
    editeur_id INT REFERENCES editeurs(id),
    image_path VARCHAR(255) DEFAULT NULL,
    image_path_url VARCHAR(255) DEFAULT NULL
);
